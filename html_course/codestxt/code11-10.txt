function iniciar() {
  var elemento = document.getElementById("canvas");
  var canvas = elemento.getContext("2d");

  canvas.beginPath();
  canvas.moveTo(100, 100);
  canvas.lineTo(200, 200);
  canvas.lineTo(100, 200);
  canvas.clip();

  canvas.beginPath();
  for (var f = 0; f < 300; f = f + 10) {
    canvas.moveTo(0, f);
    canvas.lineTo(500, f);
  }
  canvas.stroke();
}
window.addEventListener("load", iniciar);
